Edit many txt files
-------------------

# Convertir \r\n to \n (for sed to work fine)
dos2unix *
# Remove leading blank lines
find . -type f -exec sed -i '/./,$!d' {} \;

# The S. many times had many spaces after it; remove them for just 1 space
find . -type f -name '*.txt' -exec sed -i '' 's/^S\.( )+/S. /' {} \;
